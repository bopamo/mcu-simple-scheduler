#define MSS_BEGIN (   ctx)    switch(ctx) { case MSS_TASK_CTX_STATE_INIT_VAL:

MSS_BEGIN continue MSS task from last executed context. This shall be the first instruction in a MSS task function

#define MSS_FINISH ( )    }

MSS_FINISH finishing mss task context. This shall be the last instruction in a MSS task function

#define MSS_RETURN (   ctx)    do{ctx=__LINE__; return; case __LINE__:;}while(0)

MSS_RETURN suspend MSS task (give up CPU intentionally to other task) and save the task context. This function might be encapsulated by other macro function such as MSS_TIMER_WAIT_EXPIRES
